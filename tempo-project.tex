\begin{frame}
{Problem Formulation}
\begin{align}
\text{minimize}&&\int_{0}^{T} || \left[
\begin{array}{c}
x(t)\\
u(t)\\
\end{array}
\right]-y(t) ||_{W} ^{2} dt + || x(T) - y(T) ||_{W_N}^ {2} //
\text{subject to} x(0) = && \hat{x}_0,
                        && \dot{x} = f(x(t), u(t)),
                        && 0 \leq u(1) \leq 10,
                        && -7.5 \leq u(2) \leq 0.5,
                        && -7.5 \leq u(3) \leq 7.5,
                        && -7.5 \leq u(4) \leq 7.5,
                       -7.5  \ u \leq  7.5,
                        && -7.5 \leq v \leq 7.5,
                        && -7.5 \leq w \leq 7.5,
\end{align}
